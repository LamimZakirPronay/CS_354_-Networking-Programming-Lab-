string findErrorFun(int msgRec[],int len){
        int NewParity,sum=0;
        for (int i=0;i<len;i++)
                sum+=msgRec[i];
        NewParity=sum%2;
        if (msgRec[len] == NewParity)
                return "No any error of one bit (might be more than one)\n";
        else
                return "Code have some error\n";
}
